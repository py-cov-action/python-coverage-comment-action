28def checked_out_branch(git: subprocess.Git, branch: str): 

29 try: 

30 current_checkout = git.branch("--show-current").strip() 

31 except subprocess.SubProcessError: 

32 current_checkout = git.rev_parse("--short", "HEAD").strip() 

33 

34 log.debug(f"Current checkout is {current_checkout}") 

35 

36 log.debug("Resetting all changes") 

37 # Goodbye `.coverage` file. 

38 git.reset("--hard") 

39 

40 try: 

41 git.fetch("origin", branch) 

42 except subprocess.SubProcessError: 

43 # Branch seems to no exist, OR fetch failed for a different reason. 

44 # Let's make sure: 

45 # 1/ Fetch again, but this time all the remote 

46 git.fetch("origin") 

47 # 2/ And check that our branch really doesn't exist 

48 try: 

49 git.rev_parse("--verify", f"origin/{branch}") 

50 except subprocess.SubProcessError: 

51 # Ok good, the branch really doesn't exist. 

52 pass 

53 else: 

54 # Ok, our branch exist, but we failed to fetch it. Let's raise. 

55 raise 

56 log.debug(f"Branch {branch} doesn't exist.") 

57 log.info(f"Creating branch {branch}") 

58 git.switch("--orphan", branch) 

59 else: 

60 log.debug(f"Branch {branch} exist.") 

61 git.switch(branch) 

62 

63 try: 

64 yield 

65 finally: 

66 log.debug(f"Back to checkout of {current_checkout}") 

67 git.switch(current_checkout)

70def commit_operations( 

71 operations: list[files.Operation], 

72 git: subprocess.Git, 

73 branch: str, 

74): 

75 """ 

76 Store the given files. 

77 

78 Parameters 

79 ---------- 

80 operations : list[files.Operation] 

81 File operations to process 

82 git : subprocess.Git 

83 Git actor 

84 branch : str 

85 branch on which to store the files 

86 initial_file : files.Operation 

87 In case the branch didn't exist, initialize it with this initial file. 

88 """ 

89 with checked_out_branch(git=git, branch=branch): 

90 for op in operations: 

91 op.apply() 

92 git.add(str(op.path)) 

93 

94 try: 

95 git.diff("--staged", "--exit-code") 

96 except subprocess.GitError: 

97 pass # All good, command returns 1 if there's diff, 0 otherwise 

98 else: 

99 log.info("No change detected, skipping.") 

100 return 

101 

102 log.info("Saving coverage files") 

103 git.commit( 

104 "--message", 

105 GIT_COMMIT_MESSAGE, 

106 env=COMMIT_ENVIRONMENT, 

107 ) 

108 git.push("origin", branch) 

109 

110 log.info("Files saved")

113def get_datafile_contents( 

114 github: github_client.GitHub, 

115 repository: str, 

116 branch: str, 

117) -> str | None: 

118 contents_path = github.repos(repository).contents(str(files.DATA_PATH))

127def get_raw_file_url( 

128 repository: str, 

129 branch: str, 

130 path: pathlib.Path, 

131 is_public: bool, 

132): 

133 if not is_public: 

134 # If the repository is private, then the real links to raw.githubusercontents.com 

135 # will be short-lived. In this case, it's better to keep an URL that will 

136 # redirect to the correct URL just when asked. 

137 return f"https://github.com/{repository}/raw/{branch}/{path}" 

138 

139 # Otherwise, we can access the file directly. (shields.io doesn't like the 

140 # github.com domain) 

141 return f"https://raw.githubusercontent.com/{repository}/{branch}/{path}"

144 # github.repos(repository).contents(str(path)).get(ref=branch).download_url

146 # and for private repos, it would create URLs that become inactive after a few 

147 # seconds.

150def get_repo_file_url(repository: str, branch: str, path: str = "/") -> str: 

151 """ 

152 Computes the GitHub Web UI URL for a given path: 

153 If the path is empty or ends with a slash, it will be interpreted as a folder, 

154 so the URL will point to the page listing files and displaying the README. 

155 Otherwise, the URL will point to the page displaying the file contents within 

156 the UI. 

157 Leading and trailing slashes in path are removed from the final URL. 

158 """ 

159 # See test_get_repo_file_url for precise specifications 

160 path = "/" + path.lstrip("/") 

161 part = "tree" if path.endswith("/") else "blob" 

162 return f"https://github.com/{repository}/{part}/{branch}{path}".rstrip("/")
